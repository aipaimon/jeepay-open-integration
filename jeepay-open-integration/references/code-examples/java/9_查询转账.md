# 查询转账

API 路径：`api/transfer/query`

Request 类：`TransferOrderQueryRequest`
Response 类：`TransferOrderQueryResponse`

> 请求/响应参数详见 [查询转账订单文档](https://doc.jeequan.com/#/integrate/open/api/94)

---

## 响应示例

**成功响应**：

```json
{"code":0,"msg":"SUCCESS","data":{"transferId":"T202406181543441860003","mchOrderNo":"tfo1718712345678","appId":"60cc09bce4b0f1c0b83761c9","mchNo":"M1621873433953","amount":1,"currency":"CNY","state":2,"accountNo":"oMock123456","accountName":"测试用户","successTime":"2024-06-18 15:44:00"}}
```

**失败响应**：

```json
{"code":9999,"msg":"转账订单不存在"}
```

## 完整代码示例

```java
import com.jeequan.jeepay.JeepayClient;
import com.jeequan.jeepay.exception.JeepayException;
import com.jeequan.jeepay.model.TransferOrderQueryReqModel;
import com.jeequan.jeepay.request.TransferOrderQueryRequest;
import com.jeequan.jeepay.response.TransferOrderQueryResponse;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

public class TransferOrderQueryExample {

    private static final Logger log = LoggerFactory.getLogger(TransferOrderQueryExample.class);
    private static final String APP_ID = "你的应用ID";
    private static final String API_KEY = "你的API密钥";
    private static final String API_BASE = "https://你的网关地址";
    private static final String MCH_NO = "你的商户号";

    /**
     * 单例模式：同一 appId 返回同一实例
     * 多商户场景请使用 new JeepayClient(appId, apiKey, apiBase) 创建独立实例
     */
    private static JeepayClient getJeepayClient() {
        return JeepayClient.getInstance(APP_ID, API_KEY, API_BASE);
    }

    public static void main(String[] args) {
        JeepayClient client = getJeepayClient();

        TransferOrderQueryReqModel model = new TransferOrderQueryReqModel();
        model.setMchNo(MCH_NO);                                       // 商户号
        model.setAppId(client.getAppId());                            // 应用ID（从客户端获取）
        model.setTransferId("T202108121543441860003");                // 转账订单号（与mchOrderNo二选一）

        TransferOrderQueryRequest request = new TransferOrderQueryRequest();
        request.setBizModel(model);

        try {
            log.info("[Jeepay] 查询转账请求 reqUrl={} transferId={}",
                    client.getApiBase() + "/" + request.getApiUri(), model.getTransferId());

            TransferOrderQueryResponse response = client.execute(request);

            if (response.isSuccess(API_KEY)) {
                log.info("[Jeepay] 查询转账成功 code={} msg={} data={}",
                        response.getCode(), response.getMsg(),
                        com.alibaba.fastjson.JSON.toJSONString(response.get()));
                log.info("转账状态：{}", response.get().getState());         // 转账状态：0-生成，1-转账中，2-成功，3-失败
                log.info("转账金额：{}", response.get().getAmount());        // 转账金额，单位：分
            } else {
                log.warn("[Jeepay] 查询转账失败 code={} msg={}", response.getCode(), response.getMsg());
            }
        } catch (JeepayException e) {
            log.error("[Jeepay] 查询转账异常", e);
        }
    }
}
```
