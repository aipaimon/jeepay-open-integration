# 转账

API 路径：`api/transferOrder`

Request 类：`TransferOrderCreateRequest`
Response 类：`TransferOrderCreateResponse`

> 请求/响应参数详见 [发起转账订单文档](https://doc.jeequan.com/#/integrate/open/api/93)

---

## 响应示例

**成功响应**：

```json
{"code":0,"msg":"SUCCESS","data":{"transferId":"T202406181543441860003","mchOrderNo":"tfo1718712345678","state":1}}
```

**失败响应**：

```json
{"code":9999,"msg":"收款账号不存在"}
```

## 完整代码示例

```java
import com.jeequan.jeepay.JeepayClient;
import com.jeequan.jeepay.exception.JeepayException;
import com.jeequan.jeepay.model.TransferOrderCreateReqModel;
import com.jeequan.jeepay.model.TransferOrderCreateResModel;
import com.jeequan.jeepay.request.TransferOrderCreateRequest;
import com.jeequan.jeepay.response.TransferOrderCreateResponse;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

public class TransferOrderCreateExample {

    private static final Logger log = LoggerFactory.getLogger(TransferOrderCreateExample.class);
    private static final String APP_ID = "你的应用ID";
    private static final String API_KEY = "你的API密钥";
    private static final String API_BASE = "https://你的网关地址";
    private static final String MCH_NO = "你的商户号";

    /**
     * 单例模式：同一 appId 返回同一实例
     * getInstance 按 appId 缓存实例，多商户传不同 appId 即可
     */
    private static JeepayClient getJeepayClient() {
        return JeepayClient.getInstance(APP_ID, API_KEY, API_BASE);
    }

    public static void main(String[] args) {
        JeepayClient client = getJeepayClient();

        TransferOrderCreateReqModel model = new TransferOrderCreateReqModel();
        model.setMchNo(MCH_NO);                                       // 商户号
        model.setAppId(client.getAppId());                            // 应用ID（从客户端获取）
        model.setMchOrderNo("tfo" + System.currentTimeMillis());     // 商户转账单号（需唯一）
        model.setAmount(1L);                                          // 转账金额，单位：分（1分）
        model.setCurrency("CNY");                                     // 货币代码，固定CNY
        model.setIfCode("wxpay");                                     // 支付接口代码：wxpay/alipay/demomockpay
        model.setEntryType("WX_CASH");                                // 入账方式：WX_CASH-微信零钱，ALIPAY_CASH-支付宝余额
        model.setAccountNo("用户openid");                              // 收款账号（微信为openid，支付宝为用户ID）
        model.setAccountName("收款人姓名");                            // 收款人姓名（可选）
        model.setTransferDesc("测试转账");                             // 转账备注（可选）
        model.setClientIp("192.168.1.100");                           // 客户端IP地址（可选）
        model.setNotifyUrl("https://你的域名/notify/transfer");        // 转账异步通知地址（可选）

        TransferOrderCreateRequest request = new TransferOrderCreateRequest();
        request.setBizModel(model);

        try {
            log.info("[Jeepay] 转账请求 reqUrl={} ifCode={} entryType={} amount={} accountNo={}",
                    client.getApiBase() + "/" + request.getApiUri(),
                    model.getIfCode(), model.getEntryType(), model.getAmount(), model.getAccountNo());

            TransferOrderCreateResponse response = client.execute(request);

            if (response.isSuccess(API_KEY)) {
                TransferOrderCreateResModel resModel = response.get();
                log.info("[Jeepay] 转账成功 code={} msg={} data={}",
                        response.getCode(), response.getMsg(),
                        com.alibaba.fastjson.JSON.toJSONString(resModel));
                log.info("transferId: {}", resModel.getTransferId());     // 转账订单号（网关生成）
                log.info("mchOrderNo: {}", resModel.getMchOrderNo());     // 商户转账单号
            } else {
                log.warn("[Jeepay] 转账失败 code={} msg={}", response.getCode(), response.getMsg());
                if (response.get() != null) {
                    log.warn("errCode: {}, errMsg: {}", response.get().getErrCode(), response.get().getErrMsg());
                }
            }
        } catch (JeepayException e) {
            log.error("[Jeepay] 转账异常", e);
        }
    }
}
```
