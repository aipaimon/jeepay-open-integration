# 退款

API 路径：`api/refund/refundOrder`

Request 类：`RefundOrderCreateRequest`
Response 类：`RefundOrderCreateResponse`

> 请求/响应参数详见 [发起支付退款文档](https://doc.jeequan.com/#/integrate/open/api/90)

---

## 响应示例

**成功响应**：

```json
{"code":0,"msg":"SUCCESS","data":{"channelOrderNo":"DEMOMOCK_T2056570316258009089","mchRefundNo":"REF1779159547217","payAmount":100,"refundAmount":100,"refundOrderId":"R2056570316211871745","state":2}}
```

**失败响应**：

```json
{"code":9999,"msg":"退款金额超出订单可退金额"}
```

## 完整代码示例

```java
import com.jeequan.jeepay.JeepayClient;
import com.jeequan.jeepay.exception.JeepayException;
import com.jeequan.jeepay.model.RefundOrderCreateReqModel;
import com.jeequan.jeepay.model.RefundOrderCreateResModel;
import com.jeequan.jeepay.request.RefundOrderCreateRequest;
import com.jeequan.jeepay.response.RefundOrderCreateResponse;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

public class RefundOrderCreateExample {

    private static final Logger log = LoggerFactory.getLogger(RefundOrderCreateExample.class);
    private static final String APP_ID = "你的应用ID";
    private static final String API_KEY = "你的API密钥";
    private static final String API_BASE = "https://你的网关地址";
    private static final String MCH_NO = "你的商户号";

    /**
     * getInstance 按 appId 缓存实例，多商户传不同 appId 即可
     */
    private static JeepayClient getJeepayClient() {
        return JeepayClient.getInstance(APP_ID, API_KEY, API_BASE);
    }

    public static void main(String[] args) {
        JeepayClient client = getJeepayClient();

        RefundOrderCreateReqModel model = new RefundOrderCreateReqModel();
        model.setMchNo(MCH_NO);                                       // 商户号
        model.setAppId(client.getAppId());                            // 应用ID（从客户端获取）
        model.setPayOrderId("P202106181104177050002");                // 原支付订单号（与mchOrderNo二选一）
        model.setMchRefundNo("rfd" + System.currentTimeMillis());    // 商户退款单号（需唯一）
        model.setRefundAmount(1L);                                    // 退款金额，单位：分
        model.setCurrency("CNY");                                     // 货币代码，固定CNY
        model.setRefundReason("用户申请退款");                          // 退款原因（可选）
        model.setNotifyUrl("https://你的域名/notify/refund");          // 退款异步通知地址

        RefundOrderCreateRequest request = new RefundOrderCreateRequest();
        request.setBizModel(model);

        try {
            log.info("[Jeepay] 退款请求 reqUrl={} payOrderId={} refundAmount={}",
                    API_BASE + "/" + request.getApiUri(),
                    model.getPayOrderId(), model.getRefundAmount());

            RefundOrderCreateResponse response = client.execute(request);

            if (response.isSuccess(API_KEY)) {
                RefundOrderCreateResModel resModel = response.get();
                log.info("[Jeepay] 退款成功 code={} msg={} data={}",
                        response.getCode(), response.getMsg(),
                        com.alibaba.fastjson.JSON.toJSONString(resModel));
                log.info("refundOrderId: {}", resModel.getRefundOrderId());  // 退款订单号（网关生成）
                log.info("mchRefundNo: {}", resModel.getMchRefundNo());      // 商户退款单号
            } else {
                log.warn("[Jeepay] 退款失败 code={} msg={}", response.getCode(), response.getMsg());
                if (response.get() != null) {
                    log.warn("errCode: {}, errMsg: {}", response.get().getErrCode(), response.get().getErrMsg());
                }
            }
        } catch (JeepayException e) {
            log.error("[Jeepay] 退款异常", e);
        }
    }
}
```
