# 查询退款

API 路径：`api/refund/query`

Request 类：`RefundOrderQueryRequest`
Response 类：`RefundOrderQueryResponse`

> 请求/响应参数详见 [查询退款订单文档](https://doc.jeequan.com/#/integrate/open/api/91)

---

## 响应示例

**成功响应**：

```json
{"code":0,"msg":"SUCCESS","data":{"refundOrderId":"R2056570316211871745","mchRefundNo":"REF1779159547217","payOrderId":"P2056570316211871745","appId":"60cc09bce4b0f1c0b83761c9","mchNo":"M1621873433953","refundAmount":100,"currency":"CNY","state":2,"channelOrderNo":"DEMOMOCK_T2056570316258009089","createdAt":"2024-06-18 11:05:52","successTime":"2024-06-18 11:05:53"}}
```

**失败响应**：

```json
{"code":9999,"msg":"退款订单不存在"}
```

## 完整代码示例

```java
import com.jeequan.jeepay.JeepayClient;
import com.jeequan.jeepay.exception.JeepayException;
import com.jeequan.jeepay.model.RefundOrderQueryReqModel;
import com.jeequan.jeepay.request.RefundOrderQueryRequest;
import com.jeequan.jeepay.response.RefundOrderQueryResponse;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

public class RefundOrderQueryExample {

    private static final Logger log = LoggerFactory.getLogger(RefundOrderQueryExample.class);
    private static final String APP_ID = "你的应用ID";
    private static final String API_KEY = "你的API密钥";
    private static final String API_BASE = "https://你的网关地址";
    private static final String MCH_NO = "你的商户号";

    /**
     * 单例模式：同一 appId 返回同一实例
     * 多商户场景请使用 new JeepayClient(appId, apiKey, apiBase) 创建独立实例
     */
    private static JeepayClient getJeepayClient() {
        return JeepayClient.getInstance(APP_ID, API_KEY, API_BASE);
    }

    public static void main(String[] args) {
        JeepayClient client = getJeepayClient();

        RefundOrderQueryReqModel model = new RefundOrderQueryReqModel();
        model.setMchNo(MCH_NO);                                       // 商户号
        model.setAppId(client.getAppId());                            // 应用ID（从客户端获取）
        model.setRefundOrderId("P202106181105527690009");              // 退款订单号（与mchRefundNo二选一）

        RefundOrderQueryRequest request = new RefundOrderQueryRequest();
        request.setBizModel(model);

        try {
            log.info("[Jeepay] 查询退款请求 reqUrl={} refundOrderId={}",
                    API_BASE + "/" + request.getApiUri(), model.getRefundOrderId());

            RefundOrderQueryResponse response = client.execute(request);

            if (response.isSuccess(API_KEY)) {
                log.info("[Jeepay] 查询退款成功 code={} msg={} data={}",
                        response.getCode(), response.getMsg(),
                        com.alibaba.fastjson.JSON.toJSONString(response.get()));
                log.info("退款状态：{}", response.get().getState());          // 退款状态：0-生成，1-退款中，2-成功，3-失败
                log.info("退款金额：{}", response.get().getRefundAmount());   // 退款金额，单位：分
            } else {
                log.warn("[Jeepay] 查询退款失败 code={} msg={}", response.getCode(), response.getMsg());
            }
        } catch (JeepayException e) {
            log.error("[Jeepay] 查询退款异常", e);
        }
    }
}
```
