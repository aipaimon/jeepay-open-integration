# 关闭订单

API 路径：`api/pay/close`

Request 类：`PayOrderCloseRequest`
Response 类：`PayOrderCloseResponse`

> 请求/响应参数详见 [关闭支付订单文档](https://doc.jeequan.com/#/integrate/open/api/87)

---

## 响应示例

**成功响应**：

```json
{"code":0,"msg":"SUCCESS"}
```

**失败响应**：

```json
{"code":9999,"msg":"当前订单不可关闭"}
```

## 完整代码示例

```java
import com.jeequan.jeepay.JeepayClient;
import com.jeequan.jeepay.exception.JeepayException;
import com.jeequan.jeepay.model.PayOrderCloseReqModel;
import com.jeequan.jeepay.request.PayOrderCloseRequest;
import com.jeequan.jeepay.response.PayOrderCloseResponse;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

public class PayOrderCloseExample {

    private static final Logger log = LoggerFactory.getLogger(PayOrderCloseExample.class);
    private static final String APP_ID = "你的应用ID";
    private static final String API_KEY = "你的API密钥";
    private static final String API_BASE = "https://你的网关地址";
    private static final String MCH_NO = "你的商户号";

    /**
     * 单例模式：同一 appId 返回同一实例
     * 多商户场景请使用 new JeepayClient(appId, apiKey, apiBase) 创建独立实例
     */
    private static JeepayClient getJeepayClient() {
        return JeepayClient.getInstance(APP_ID, API_KEY, API_BASE);
    }

    public static void main(String[] args) {
        JeepayClient client = getJeepayClient();

        PayOrderCloseReqModel model = new PayOrderCloseReqModel();
        model.setMchNo(MCH_NO);                                       // 商户号
        model.setAppId(client.getAppId());                            // 应用ID（从客户端获取）
        model.setPayOrderId("P1485879219030011906");                  // 要关闭的支付订单号（与mchOrderNo二选一）

        PayOrderCloseRequest request = new PayOrderCloseRequest();
        request.setBizModel(model);

        try {
            log.info("[Jeepay] 关闭订单请求 reqUrl={} payOrderId={}",
                    API_BASE + "/" + request.getApiUri(), model.getPayOrderId());

            PayOrderCloseResponse response = client.execute(request);

            if (response.isSuccess(API_KEY)) {
                log.info("[Jeepay] 关闭订单成功 code={} msg={}", response.getCode(), response.getMsg());
            } else {
                log.warn("[Jeepay] 关闭订单失败 code={} msg={}", response.getCode(), response.getMsg());
            }
        } catch (JeepayException e) {
            log.error("[Jeepay] 关闭订单异常", e);
        }
    }
}
```
