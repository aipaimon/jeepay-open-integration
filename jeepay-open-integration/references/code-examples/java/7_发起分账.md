# 发起订单分账

API 路径：`api/division/exec`

Request 类：`PayOrderDivisionExecRequest`
Response 类：`PayOrderDivisionExecResponse`

> 请求/响应参数详见 [发起订单分账文档](https://doc.jeequan.com/#/integrate/open/api/97)

---

## 响应示例

**成功响应**：

```json
{"code":0,"msg":"SUCCESS","data":{"channelBatchOrderId":"4200001234202406181234567890"}}
```

**失败响应**：

```json
{"code":9999,"msg":"订单未支付成功，不允许分账"}
```

## 完整代码示例

```java
import com.alibaba.fastjson.JSONArray;
import com.alibaba.fastjson.JSONObject;
import com.jeequan.jeepay.JeepayClient;
import com.jeequan.jeepay.exception.JeepayException;
import com.jeequan.jeepay.model.PayOrderDivisionExecReqModel;
import com.jeequan.jeepay.request.PayOrderDivisionExecRequest;
import com.jeequan.jeepay.response.PayOrderDivisionExecResponse;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

public class PayOrderDivisionExecExample {

    private static final Logger log = LoggerFactory.getLogger(PayOrderDivisionExecExample.class);
    private static final String APP_ID = "你的应用ID";
    private static final String API_KEY = "你的API密钥";
    private static final String API_BASE = "https://你的网关地址";
    private static final String MCH_NO = "你的商户号";

    /**
     * 单例模式：同一 appId 返回同一实例
     * 多商户场景请使用 new JeepayClient(appId, apiKey, apiBase) 创建独立实例
     */
    private static JeepayClient getJeepayClient() {
        return JeepayClient.getInstance(APP_ID, API_KEY, API_BASE);
    }

    public static void main(String[] args) {
        JeepayClient client = getJeepayClient();

        PayOrderDivisionExecReqModel model = new PayOrderDivisionExecReqModel();
        model.setMchNo(MCH_NO);                                       // 商户号
        model.setAppId(client.getAppId());                            // 应用ID（从客户端获取）
        model.setPayOrderId("P1470667876906389505");                  // 支付订单号（与mchOrderNo二选一）

        // 方式一：使用系统自动分账组
        // model.setUseSysAutoDivisionReceivers((byte) 1);            // 使用系统自动分账组：1-是

        // 方式二：手动指定接收者
        model.setUseSysAutoDivisionReceivers((byte) 0);               // 使用系统自动分账组：0-否（手动指定）
        JSONArray receivers = new JSONArray();
        receivers.add(JSONObject.parseObject("{receiverId: '800004', receiverGroupId: '', divisionProfit: '0.1'}"));
        receivers.add(JSONObject.parseObject("{receiverId: '800005', receiverGroupId: '', divisionProfit: '0.2'}"));
        model.setReceivers(receivers.toJSONString());                 // 接收者列表JSON

        PayOrderDivisionExecRequest request = new PayOrderDivisionExecRequest();
        request.setBizModel(model);

        try {
            log.info("[Jeepay] 发起分账请求 reqUrl={} payOrderId={}",
                    client.getApiBase() + "/" + request.getApiUri(), model.getPayOrderId());

            PayOrderDivisionExecResponse response = client.execute(request);

            if (response.isSuccess(API_KEY)) {
                log.info("[Jeepay] 分账成功 code={} msg={} data={}",
                        response.getCode(), response.getMsg(),
                        com.alibaba.fastjson.JSON.toJSONString(response.get()));
                log.info("渠道分账订单号：{}", response.get().getChannelBatchOrderId());
            } else {
                log.warn("[Jeepay] 分账失败 code={} msg={}", response.getCode(), response.getMsg());
                if (response.get() != null) {
                    log.warn("errCode: {}, errMsg: {}", response.get().getErrCode(), response.get().getErrMsg());
                }
            }
        } catch (JeepayException e) {
            log.error("[Jeepay] 分账异常", e);
        }
    }
}
```
