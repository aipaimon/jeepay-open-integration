# 绑定分账用户

API 路径：`api/division/receiver/bind`

Request 类：`DivisionReceiverBindRequest`
Response 类：`DivisionReceiverBindResponse`

> 请求/响应参数详见 [绑定分账用户文档](https://doc.jeequan.com/#/integrate/open/api/96)

---

## 响应示例

**成功响应**：

```json
{"code":0,"msg":"SUCCESS","data":{"receiverId":"800004","accNo":"oMock123456","accName":"测试商户","bindState":1}}
```

**失败响应**：

```json
{"code":9999,"msg":"接收账号已绑定"}
```

## 完整代码示例

```java
import com.jeequan.jeepay.JeepayClient;
import com.jeequan.jeepay.exception.JeepayException;
import com.jeequan.jeepay.model.DivisionReceiverBindReqModel;
import com.jeequan.jeepay.request.DivisionReceiverBindRequest;
import com.jeequan.jeepay.response.DivisionReceiverBindResponse;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

public class DivisionReceiverBindExample {

    private static final Logger log = LoggerFactory.getLogger(DivisionReceiverBindExample.class);
    private static final String APP_ID = "你的应用ID";
    private static final String API_KEY = "你的API密钥";
    private static final String API_BASE = "https://你的网关地址";
    private static final String MCH_NO = "你的商户号";

    /**
     * 单例模式：同一 appId 返回同一实例
     * getInstance 按 appId 缓存实例，多商户传不同 appId 即可
     */
    private static JeepayClient getJeepayClient() {
        return JeepayClient.getInstance(APP_ID, API_KEY, API_BASE);
    }

    public static void main(String[] args) {
        JeepayClient client = getJeepayClient();

        DivisionReceiverBindReqModel model = new DivisionReceiverBindReqModel();
        model.setMchNo(MCH_NO);                                       // 商户号
        model.setAppId(client.getAppId());                            // 应用ID（从客户端获取）
        model.setIfCode("wxpay");                                     // 支付接口代码：wxpay/alipay/demomockpay
        model.setAccType((byte) 1);                                   // 账号类型：0-个人(对私)，1-商户(对公)
        model.setAccNo("接收账号");                                    // 分账接收账号（如微信openid）
        model.setAccName("接收人名称");                                // 账号持有人名称
        model.setRelationType("SERVICE_PROVIDER");                    // 关系类型：SERVICE_PROVIDER/STORE/STAFF/PARTNER等
        model.setRelationTypeName("服务商");                           // 关系类型名称（可选）
        model.setDivisionProfit("0.10");                              // 分账比例，"0.10"表示10%

        DivisionReceiverBindRequest request = new DivisionReceiverBindRequest();
        request.setBizModel(model);

        try {
            log.info("[Jeepay] 绑定分账用户请求 reqUrl={} ifCode={} accNo={}",
                    client.getApiBase() + "/" + request.getApiUri(), model.getIfCode(), model.getAccNo());

            DivisionReceiverBindResponse response = client.execute(request);

            if (response.isSuccess(API_KEY)) {
                log.info("[Jeepay] 绑定分账用户成功 code={} msg={} data={}",
                        response.getCode(), response.getMsg(),
                        com.alibaba.fastjson.JSON.toJSONString(response.get()));
                log.info("accNo: {}, 绑定成功", response.get().getAccNo());
            } else {
                log.warn("[Jeepay] 绑定分账用户失败 code={} msg={}", response.getCode(), response.getMsg());
                if (response.get() != null) {
                    log.warn("errCode: {}, errMsg: {}", response.get().getErrCode(), response.get().getErrMsg());
                }
            }
        } catch (JeepayException e) {
            log.error("[Jeepay] 绑定分账用户异常", e);
        }
    }
}
```
