# 查询订单

API 路径：`api/pay/query`

Request 类：`PayOrderQueryRequest`
Response 类：`PayOrderQueryResponse`

> 请求/响应参数详见 [查询支付订单文档](https://doc.jeequan.com/#/integrate/open/api/86)

---

## 响应示例

**成功响应**：

```json
{"code":0,"msg":"SUCCESS","data":{"payOrderId":"P2056570316211871745","mchOrderNo":"mho1718712345678","appId":"60cc09bce4b0f1c0b83761c9","mchNo":"M1621873433953","wayCode":"WX_NATIVE","amount":1,"currency":"CNY","orderState":2,"clientIp":"192.168.1.100","subject":"测试商品","createdAt":"2024-06-18 11:04:17","successTime":"2024-06-18 11:05:30"}}
```

**失败响应**：

```json
{"code":9999,"msg":"订单不存在"}
```

## 完整代码示例

```java
import com.jeequan.jeepay.JeepayClient;
import com.jeequan.jeepay.exception.JeepayException;
import com.jeequan.jeepay.model.PayOrderQueryReqModel;
import com.jeequan.jeepay.request.PayOrderQueryRequest;
import com.jeequan.jeepay.response.PayOrderQueryResponse;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

public class PayOrderQueryExample {

    private static final Logger log = LoggerFactory.getLogger(PayOrderQueryExample.class);
    private static final String APP_ID = "你的应用ID";
    private static final String API_KEY = "你的API密钥";
    private static final String API_BASE = "https://你的网关地址";
    private static final String MCH_NO = "你的商户号";

    /**
     * 单例模式：同一 appId 返回同一实例
     * 多商户场景请使用 new JeepayClient(appId, apiKey, apiBase) 创建独立实例
     */
    private static JeepayClient getJeepayClient() {
        return JeepayClient.getInstance(APP_ID, API_KEY, API_BASE);
    }

    public static void main(String[] args) {
        JeepayClient client = getJeepayClient();

        PayOrderQueryReqModel model = new PayOrderQueryReqModel();
        model.setMchNo(MCH_NO);                                       // 商户号
        model.setAppId(client.getAppId());                            // 应用ID（从客户端获取）
        model.setPayOrderId("P202106181104177050002");                // 支付订单号（与mchOrderNo二选一）

        PayOrderQueryRequest request = new PayOrderQueryRequest();
        request.setBizModel(model);

        try {
            log.info("[Jeepay] 查询订单请求 reqUrl={} payOrderId={}",
                    API_BASE + "/" + request.getApiUri(), model.getPayOrderId());

            PayOrderQueryResponse response = client.execute(request);

            if (response.isSuccess(API_KEY)) {
                log.info("[Jeepay] 查询订单成功 code={} msg={} data={}",
                        response.getCode(), response.getMsg(),
                        com.alibaba.fastjson.JSON.toJSONString(response.get()));
                log.info("金额：{}", response.get().getAmount());     // 订单金额，单位：分
            } else {
                log.warn("[Jeepay] 查询订单失败 code={} msg={}", response.getCode(), response.getMsg());
            }
        } catch (JeepayException e) {
            log.error("[Jeepay] 查询订单异常", e);
        }
    }
}
```
