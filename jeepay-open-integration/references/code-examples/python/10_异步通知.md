# 异步通知验签处理

适用接口：支付通知、退款通知、转账通知

SDK 工具：`jeepay.utils.Utils._signature_rules(params)`

---

## 签名算法

Jeepay 使用 MD5 签名，流程：

1. 将通知参数中 `sign` 字段移除
2. 剩余参数按 key 的 ASCII 码升序排列
3. 拼接为 `key1=value1&key2=value2&...&key=apiKey`
4. 对拼接字符串做 MD5，结果转大写

SDK 工具方法：`jeepay.utils.Utils._signature_rules(params)` 可直接计算签名。

## 通知参数说明

异步通知的参数是**非固定的**，Jeepay 可能会在文档中增加新的通知字段。必须使用动态解析方式获取参数，不要硬编码字段。

## Flask 完整处理示例

```python
import logging
from flask import Flask, request

from jeepay.app_config import AppConfig
from jeepay.utils import Utils

app = Flask(__name__)
logging.basicConfig(level=logging.INFO)
log = logging.getLogger(__name__)

# API 密钥（与 SDK 配置一致）
API_KEY = "你的API密钥"


def verify_sign(params):
    """
    验证通知签名
    使用 SDK 的 Utils._signature_rules() 方法计算签名，与通知中的 sign 对比
    """
    sign = params.get("sign")
    if not sign:
        log.warning("[Jeepay] 通知签名为空")
        return False

    # 移除 sign 字段后计算签名
    sign_params = {k: v for k, v in params.items() if k != "sign"}
    computed_sign = Utils._signature_rules(sign_params)
    return sign.upper() == computed_sign.upper()


@app.route("/notify/pay", methods=["POST"])
def pay_notify():
    """支付异步通知"""
    # Step 1: 动态解析通知参数（不硬编码字段，向前兼容）
    params = dict(request.form)
    log.info("[Jeepay] 支付通知参数: %s", params)

    # Step 2: 验签
    if not verify_sign(params):
        log.warning("[Jeepay] 通知验签失败")
        return "fail"

    # Step 3: 幂等处理（同一订单可能收到多次通知）
    mch_order_no = params.get("mchOrderNo")
    # order = order_service.find_by_mch_order_no(mch_order_no)
    # if order and order.is_processed:
    #     return "success"

    # Step 4: 业务处理
    pay_order_id = params.get("payOrderId")
    state = params.get("state")

    if state == "2":
        log.info("[Jeepay] 支付成功 payOrderId=%s mchOrderNo=%s", pay_order_id, mch_order_no)
        # 更新本地订单状态...
    else:
        log.info("[Jeepay] 支付状态 state=%s payOrderId=%s", state, pay_order_id)

    # Step 5: 返回成功标识（Jeepay 收到 "success" 后停止重试）
    return "success"


@app.route("/notify/refund", methods=["POST"])
def refund_notify():
    """退款异步通知"""
    params = dict(request.form)
    log.info("[Jeepay] 退款通知参数: %s", params)

    if not verify_sign(params):
        log.warning("[Jeepay] 通知验签失败")
        return "fail"

    # 业务处理
    refund_order_id = params.get("refundOrderId")
    mch_refund_no = params.get("mchRefundNo")
    state = params.get("state")

    if state == "2":
        log.info("[Jeepay] 退款成功 refundOrderId=%s mchRefundNo=%s", refund_order_id, mch_refund_no)
        # 更新本地退款状态...

    return "success"


@app.route("/notify/transfer", methods=["POST"])
def transfer_notify():
    """转账异步通知"""
    params = dict(request.form)
    log.info("[Jeepay] 转账通知参数: %s", params)

    if not verify_sign(params):
        log.warning("[Jeepay] 通知验签失败")
        return "fail"

    # 业务处理
    transfer_id = params.get("transferId")
    mch_order_no = params.get("mchOrderNo")
    state = params.get("state")

    if state == "2":
        log.info("[Jeepay] 转账成功 transferId=%s mchOrderNo=%s", transfer_id, mch_order_no)
        # 更新本地转账状态...

    return "success"


if __name__ == "__main__":
    app.run(host="0.0.0.0", port=5000)
```

## 验签核心逻辑

```python
from jeepay.utils import Utils

# _signature_rules() 内部实现：
# 1. 过滤空值，拼接为 key1=value1&key2=value2&
# 2. 按 ASCII 升序排列
# 3. 末尾追加 key=apiKey
# 4. MD5 后转大写

sign_params = {
    "appId": "...",
    "mchNo": "...",
    "payOrderId": "...",
    "state": "2",
    # ... 其他参数（不含 sign）
}

computed_sign = Utils._signature_rules(sign_params)
is_valid = notify_sign.upper() == computed_sign.upper()
```
